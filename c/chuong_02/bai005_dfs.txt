#include <stdio.h>
#include <stdlib.h>

const int inf = (int) 1e9;

void de_quy(int, int, int []);
void dfs(int, int, int [], int []);

int main() {
    int n;
    scanf("%d ", &n);

    int cnt = 0;
    int can[] = {inf};

    de_quy(n, cnt, can);

    int m = can[0];

    int *arr = (int*) calloc(m, sizeof(int));

    dfs(n, cnt, can, arr);

    for(int i = 0; i < m; i++) {
        printf("%d ", arr[i]);
    }

    free(arr);
    return 0;
}

void de_quy(int n, int cnt, int can[]) {
    if(can[0] < cnt) {
        return;
    }

    if(n == 1) {
        can[0] = cnt;
        return;
    }

    de_quy(n - 1, cnt + 1, can);

    if(n % 2 == 0) {
        de_quy(n / 2, cnt + 1, can);
    }

    if(n % 3 == 0) {
        de_quy(n / 3, cnt + 1, can);
    }
}

void dfs(int n, int cnt, int can[], int arr[]) {
    if(can[0] < cnt) {
        return;
    }

    if(n == 1) {
        can[0] = 0;
        return;
    }

    arr[cnt] = n;

    dfs(n - 1, cnt + 1, can, arr);

    if(n % 2 == 0) {
        dfs(n / 2, cnt + 1, can, arr);
    }

    if(n % 3 == 0) {
        dfs(n / 3, cnt + 1, can, arr);
    }
}
