#include <stdio.h>

const int inf = (int) 1e9;

void de_quy(int, int, int []);

int main() {
    int n;
    scanf("%d ", &n);

    int cnt = 0;
    int can[] = {inf};

    de_quy(n, cnt, can);

    printf("%d ", can[0]);
    return 0;
}

void de_quy(int n, int cnt, int can[]) {
    if(can[0] == cnt) {
        return;
    }

    if(n == 1) {
        can[0] = cnt;
        return;
    }

    de_quy(n - 1, cnt + 1, can);

    if(n % 2 == 0) {
        de_quy(n / 2, cnt + 1, can);
    }

    if(n % 3 == 0) {
        de_quy(n / 3, cnt + 1, can);
    }
}
